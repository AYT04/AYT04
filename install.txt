# --- Pre-chroot (run on installer) ---
iwctl
efivar -l
gdisk
cgdisk

mkfs.fat -F32 p1
mkswap p2
swapon p2
mkfs.ext4 p3
mkfs.ext4 p4

mount p3 /mnt
mkdir -p /mnt/boot /mnt/home
mount p1 /mnt/boot
mount p4 /mnt/home

cp mirrorlist /mnt/etc/pacman.d/mirrorlist
pacstrap -K /mnt base linux linux-firmware
genfstab -U -p /mnt >> /mnt/etc/fstab
nano /mnt/etc/fstab

arch-chroot /mnt

# --- Chroot (run inside arch-chroot) ---
ln -sf /usr/share/zoneinfo/Canada/Eastern /etc/localtime
hwclock --systohc --utc

echo "en_US.UTF-8 UTF-8" > /etc/locale.gen
locale-gen
echo "LANG=en_US.UTF-8" > /etc/locale.conf

echo "Overlord" > /etc/hostname

systemctl enable fstrim.timer

pacman -S --noconfirm nano bash-completion sudo

# Edit /etc/pacman.conf and ensure:
# [multilib]
# Include = /etc/pacman.d/mirrorlist
nano /etc/pacman.conf

pacman -Sy --noconfirm

useradd -m -g users -G wheel,storage,power -s /bin/bash addison
passwd addison

EDITOR=nano visudo
# In visudo, ensure this line exists:
# %wheel ALL=(ALL:ALL) ALL

mount -t efivarfs efivarfs /sys/firmware/efi/efivars
bootctl install

nano /boot/loader/entries/arch.conf
# title Arch
# linux /vmlinuz-linux
# initrd /initramfs-linux.img

echo "options root=PARTUUID=$(blkid -s PARTUUID -o value /dev/p3) rw" >> /boot/loader/entries/arch.conf

pacman -S --noconfirm dhcpcd networkmanager
systemctl enable dhcpcd@wlan0.service
systemctl enable NetworkManager.service

exit

# --- Finish ---
umount -R /mnt
reboot

# --- After reboot ---
pacman -S xorg-server xorg-xinit xterm xorg-apps sddm --noconfirm
startx

pacman -S sddm --noconfirm
systemctl enable sddm.service
reboot

pacman -S steam firefox discord kate vlc --noconfirm
